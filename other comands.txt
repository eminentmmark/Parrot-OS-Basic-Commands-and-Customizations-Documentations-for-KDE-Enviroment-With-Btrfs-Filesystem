// setting up printers (epson)

		# Install printing support packages (Most Debian-based systems use CUPS for printing)
		
			sudo apt install cups cups-client printer-driver-escpr system-config-printer -y
		
		# for scanner support
		
			sudo apt install sane-airscan sane-utils xsane -y
		
		# Enable and start the printing service
		
			sudo systemctl enable cups
			sudo systemctl start cups
		
			# check status (expected output something like 'active (running)')
			
			systemctl status cups
			
		# add the user to the printer admin group
		
			sudo usermod -aG lpadmin $USER
		
		# Connect the Epson L3250
		Option A — USB Setup (Recommended First)
		Turn on the printer.
		Connect it via USB.
		Run:
			lsusb
		
		(expected output a list of [Epson] printers)
		
		Option B — Wi-Fi Setup
		
		Before Linux can discover it:
		
		Connect the printer to the same Wi-Fi network as the laptop.
		Use the Epson Smart Panel app from a phone if needed for initial Wi-Fi configuration.
		
		# After it joins Wi-Fi, continue below.
		
			Open the Linux printer manager
		
			In KDE:
		
			Open Settings
			Go to Printers
		
			OR run:
		
				system-config-printer
				
			
			# Add the printer
			
			Linux should automatically detect:
			
			Epson L3250 USB
			OR
			Epson L3250 Network Printer
			
			Click:
			
			Add
			Select the printer
			Continue
			
			If prompted for a driver:
			
			Choose:
			“Epson ESC/P-R”
			or “Recommended Driver”
			
			Finish setup.
			
			
			# Print a test page
			
			After setup:
			
			Right-click the printer
			Choose:
			Properties
			Print Test Page
			
			If the page prints, setup succeeded.

				skanpage

// setting up root for snapper configurations

			//  verify u are using Btrfs filesystem (run):
				
				findmnt -no FSTYPE /
				
			# expected resutl Btrfs
			
			run:
				sudo apt install snapper btrfs-progs -y
				
			# create proper Btrfs subvolume:
				
				sudo btrfs subvolume create /.snapshots
				
			# delete folder is it exists
			
				sudo rm -rf /.snapshots
				
			# Create the Snapper Configuration (run):
				
				sudo snapper -c root create-config /
				
			# This creates:
				the root config,
				snapshot directories,
				default settings.
				
			# change the permissions of the .snapshots directory to root
			
				sudo chmod 750 /.snapshots
				
			// create a snapshot:

				sudo snapper -c root create --description "First snapshot"

			# List snapshots:

				sudo snapper -c root list
				
						
						
						
						
						
// backups with rsync


						# create backup location

						mkdir -p /mnt/backup/system-config		{or u can use		mkdir -p /media/blair/backup/system-config}
						
						
						# basic system config backup

						sudo rsync -aAXv /etc /home /root /mnt/backup/system-config


						# Backup only configuration files (lighter)

						sudo rsync -a /etc /home/blair/.config /mnt/backup/system-config

						
						# Exclude unnecessary files (recommended)
						
						sudo rsync -aAXv \
						--exclude={"/dev/*","/proc/*","/sys/*","/tmp/*","/run/*","/mnt/*","/media/*","/lost+found"} \
						/ /mnt/backup/system-config
						
						
						# Verify backup integrity
						
						ls -lah /mnt/backup/system-config
						
						
						# restore later (usually for live USB)
						
						sudo rsync -aAXv /mnt/backup/system-config/ mnt/
