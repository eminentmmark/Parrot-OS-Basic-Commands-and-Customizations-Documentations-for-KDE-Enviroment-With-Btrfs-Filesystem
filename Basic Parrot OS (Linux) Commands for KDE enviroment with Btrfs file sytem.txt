hibernate			systemctl hibernate

hibernate function		# Create a Btrfs-compatible swap file

						sudo truncate -s 0 /swapfile
						sudo chattr +C /swapfile
						sudo dd if=/dev/zero of=/swapfile bs=1M count=32642 status=progress
						
						
						# Secure the swap file
						
						sudo chmod 600 /swapfile
					
						
						# Format and activate swap
						
						sudo mkswap /swapfile
						sudo swapon /swapfile
					
						
						# Make swap persistent across reboots
						
						echo '/swapfile none swap defaults 0 0' | sudo tee -a /etc/fstab
						
						
						# Get the filesystem UUID (that will be used in the nano file)
						
						findmnt -no UUID -T /swapfile
					
						
						# Get the Btrfs swapfile resume offset (that will be used in the nano file)
						
						sudo btrfs inspect-internal map-swapfile -r /swapfile
					
						
						# Edit GRUB configuration
						sudo nano /etc/default/grub
						
						
						//then replace:
						
						GRUB_CMDLINE_LINUX_DEFAULT="quiet splash"
						
						//with:
						
						GRUB_CMDLINE_LINUX_DEFAULT="quiet splash resume=UUID=a5850804-64f8-4f76-8f8b-c638cfba5247 resume_offset=8398080"
						
						//continue with
						
						# Update GRUB
						
						sudo update-grub
						
						
						# Rebuild initramfs (critical final fix)
						
						sudo update-initramfs -u -k all
						
						
						# Reboot
						
						sudo reboot
						
						//all set


// backups with rsync


						# create backup location

						mkdir -p /mnt/backup/system-config		{or u can use		mkdir -p /media/blair/backup/system-config}
						
						
						# basic system config backup

						sudo rsync -aAXv /etc /home /root /mnt/backup/system-config


						# Backup only configuration files (lighter)

						sudo rsync -a /etc /home/blair/.config /mnt/backup/system-config

						
						# Exclude unnecessary files (recommended)
						
						sudo rsync -aAXv \
						--exclude={"/dev/*","/proc/*","/sys/*","/tmp/*","/run/*","/mnt/*","/media/*","/lost+found"} \
						/ /mnt/backup/system-config
						
						
						# Verify backup integrity
						
						ls -lah /mnt/backup/system-config
						
						
						# restore later (usually for live USB)
						
						sudo rsync -aAXv /mnt/backup/system-config/ mnt/
						
						
						sudo update-grub
						sudo grub-install
