hibernate function		# Create a Btrfs-compatible swap file

						sudo truncate -s 0 /swapfile
						sudo chattr +C /swapfile
						sudo dd if=/dev/zero of=/swapfile bs=1M count=8192 status=progress
						
						
						# Secure the swap file
						
						sudo chmod 600 /swapfile
					
						
						# Format and activate swap
						
						sudo mkswap /swapfile
						sudo swapon /swapfile
					
						
						# Make swap persistent across reboots
						
						echo '/swapfile none swap defaults 0 0' | sudo tee -a /etc/fstab
						
						
						# Get the filesystem UUID
						
						findmnt -no UUID -T /swapfile
					
						
						# Get the Btrfs swapfile resume offset
						
						sudo btrfs inspect-internal map-swapfile -r /swapfile
					
						
						# Edit GRUB configuration
						sudo nano /etc/default/grub
						
						
						//the replace:
						
						GRUB_CMDLINE_LINUX_DEFAULT="quiet splash"
						
						//with:
						
						GRUB_CMDLINE_LINUX_DEFAULT="quiet splash resume=UUID=a5850804-64f8-4f76-8f8b-c638cfba5247 resume_offset=8398080"
						
						//continue with
						
						# Update GRUB
						
						sudo update-grub
						
						
						# Rebuild initramfs (critical final fix)
						
						sudo update-initramfs -u -k all
						
						
						# Reboot
						
						sudo reboot
						
						//all set
